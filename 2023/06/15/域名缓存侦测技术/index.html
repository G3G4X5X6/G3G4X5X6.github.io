域名缓存侦测技术

发表于 2023-06-15 分类于 信息收集

域名缓存探测技术是一种用于获取目标系统 DNS 缓存信息的技术。通过发送特定的 DNS 查询请求并观察响应时间和内容，可以推断出目标系统是否存在特定域名的 DNS 缓存记录。

例如，测试人员可以提交某安全软件更新所使用的域名，如果有记录，说明该网络使用该种安全软件。

import socket
import dns.message
import dns.rdatatype


def resolve_dns(domain, nameserver=None):
    # 创建 UDP 套接字
    sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)

    # 设置超时时间
    sock.settimeout(5)

    # 构造 DNS 请求数据包
    query = dns.message.make_query(domain, dns.rdatatype.A, dns.rdataclass.IN)
    query.flags &= ~dns.flags.RD
    # query.flags |= dns.flags.RD

    # # 添加 DNS 服务器记录
    # if nameserver:
    #     query.additional.append(
    #         dns.rrset.from_text(".", "IN", "NS", nameserver))

    # 发送 DNS 请求数据包
    sock.sendto(query.to_wire(), ("192.168.233.133", 53))

    try:
        # 接收 DNS 响应数据包
        data, _ = sock.recvfrom(1024)

        # 解析 DNS 响应数据包
        response = dns.message.from_wire(data)

        # 获取解析结果
        answers = response.answer
        print(answers)
        # for ans in answers:
        #     if ans.rdtype == dns.rdatatype.A:
        #         ips = [rdata.address for rdata in ans]
        #         return ips
    except socket.timeout:
        print("DNS query timeout")
    finally:
        # 关闭套接字
        sock.close()


# 解析 example.com 的 A 记录
ips = resolve_dns("example.com")
